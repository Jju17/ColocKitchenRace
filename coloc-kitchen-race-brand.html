    static let ckrLavender  = Color(hex: "#8B7FC7")
    static let ckrCoral     = Color(hex: "#E07A6B")
    static let ckrGold      = Color(hex: "#D4A843")

    static let ckrCoralLight    = Color(hex: "#FFD9D2")
    static let ckrGoldLight     = Color(hex: "#FFF3C4")
}
